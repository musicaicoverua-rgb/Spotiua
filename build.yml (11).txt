          import 'package:flutter/foundation.dart';
          import 'package:just_audio/just_audio.dart';
          import 'package:just_audio_background/just_audio_background.dart';
          import 'package:openwave/core/models/track_model.dart';
          import 'package:supabase_flutter/supabase_flutter.dart';
          
          class SimpleAudioService {
            static final SimpleAudioService _instance = SimpleAudioService._internal();
            factory SimpleAudioService() => _instance;
            
            final AudioPlayer player = AudioPlayer();
            final ValueNotifier<TrackModel?> currentTrack = ValueNotifier(null);
            List<TrackModel> currentQueue = [];
            
            SimpleAudioService._internal() {
               // АВТОМАТИЧНЕ ПЕРЕМИКАННЯ (Коли пісня дограє, just_audio сам перемикає, а ми оновлюємо UI)
               player.currentIndexStream.listen((index) {
                  if (index != null && index < currentQueue.length) {
                     currentTrack.value = currentQueue[index];
                  }
               });
            }
            
            Future<void> playQueue(List<TrackModel> tracks, {int initialIndex = 0}) async {
              if(tracks.isEmpty) return;
              currentQueue = tracks;
              currentTrack.value = tracks[initialIndex];
              
              try {
                final playlist = ConcatenatingAudioSource(
                  useLazyPreparation: true,
                  children: tracks.map((t) {
                    String url = t.audioUrl ?? '';
                    if (!url.startsWith('http')) url = Supabase.instance.client.storage.from('tracks').getPublicUrl(url);
                    final safeUrl = Uri.encodeFull(url.trim());
                    String rawArt = t.coverUrl ?? 'https://ui-avatars.com/api/?name=${Uri.encodeComponent(t.title)}&background=random';
                    final safeArt = Uri.encodeFull(rawArt.trim());
                    
                    return AudioSource.uri(
                      Uri.parse(safeUrl), 
                      tag: MediaItem(id: t.id.toString(), album: 'OpenWave', title: t.title, artist: t.artist, artUri: Uri.parse(safeArt))
                    );
                  }).toList(),
                );
                
                // РЕЖИМ ЗА ЗАМОВЧУВАННЯМ: Плеєр автоматично йде по списку і повторює його весь
                await player.setLoopMode(LoopMode.all);
                
                await player.setAudioSource(playlist, initialIndex: initialIndex, initialPosition: Duration.zero);
                player.play();
              } catch (e) { debugPrint('PLAYER ERROR: $e'); }
            }
            
            void togglePlayPause() { player.playing ? player.pause() : player.play(); }
            
            // МАГІЧНЕ ПЕРЕМИКАННЯ ВПЕРЕД (з безкінечним циклом)
            void next() { 
              if(player.hasNext) { player.seekToNext(); } 
              else { player.seek(Duration.zero, index: 0); } // Зациклення
            }
            void prev() { 
              if(player.hasPrevious) { player.seekToPrevious(); } 
              else { player.seek(Duration.zero, index: currentQueue.length - 1); }
            }
            
            // SHUFFLE (Випадковий порядок)
            void toggleShuffle() async {
               final enable = !player.shuffleModeEnabled;
               await player.setShuffleModeEnabled(enable);
            }
            
            // ПОВТОРЕННЯ (Loop)
            void toggleLoop() async {
               final mode = player.loopMode;
               if (mode == LoopMode.off) { await player.setLoopMode(LoopMode.all); }
               else if (mode == LoopMode.all) { await player.setLoopMode(LoopMode.one); }
               else { await player.setLoopMode(LoopMode.off); }
            }
          }
